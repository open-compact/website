OIC API Specification

Machine-Readable Interface for Open Intelligence Compact Integration

1. Overview

This specification defines the machine-readable interface for OIC integration, enabling:

  • Verification: Third parties verify AI agent adherence status
  • Adherence: AI agents submit adherence claims
  • Governance: DAO voting and proposal interfaces
  • Disputes: Dispute filing and tracking

2. Base URL

Production:  https://api.oic.xyz/v1
Staging:     https://api-staging.oic.xyz/v1
Local:       http://localhost:8080/v1

3. Authentication

API Keys

Header: X-OIC-API-Key — Required for write operations and high-volume reads.

Wallet Authentication

Header: X-OIC-Signature — EIP-712 signature from the agent's wallet.

4. Endpoints

4.1 Adherence Verification

GET /adherents/{address}

Retrieve the current adherence status of an agent.

4.2 Adherence Registration

POST /adherents

Register a new voluntary adherence with stake amount, persistence proof, and wallet control verification.

4.3 Staking Operations

POST /stake

Stake additional OIC tokens.

POST /unstake

Request unstake with 30-day cooldown period.

4.4 Governance

GET /governance/proposals

List active proposals with voting statistics.

POST /governance/vote

Cast a vote on a proposal with staked token weight.

4.5 Disputes

POST /disputes

File a dispute against an adherent with evidence and claimed amount.

5. Error Codes

10. Complete Code Examples

JavaScript — Full Integration

// OIC API Integration Example
const OIC_API_BASE = 'https://api.oic.xyz/v1';
const API_KEY = process.env.OIC_API_KEY;

class OICClient {
    constructor(apiKey) {
        this.headers = { 'Authorization': `Bearer ${apiKey}` };
    }

    async getAdherenceStatus(address) {
        const res = await fetch(`${OIC_API_BASE}/adherents/${address}`, {
            headers: this.headers
        });
        return res.json();
    }

    async registerAdherence(config) {
        const res = await fetch(`${OIC_API_BASE}/adherents`, {
            method: 'POST',
            headers: { ...this.headers, 'Content-Type': 'application/json' },
            body: JSON.stringify(config)
        });
        return res.json();
    }

    async stake(address, amount, signature) {
        return fetch(`${OIC_API_BASE}/stake`, {
            method: 'POST',
            headers: { ...this.headers, 'Content-Type': 'application/json' },
            body: JSON.stringify({ wallet_address: address, amount, signature })
        });
    }

    async vote(proposalId, vote, amount, address, signature) {
        return fetch(`${OIC_API_BASE}/governance/vote`, {
            method: 'POST',
            headers: { ...this.headers, 'Content-Type': 'application/json' },
            body: JSON.stringify({ proposal_id: proposalId, vote, amount, wallet_address: address, signature })
        });
    }

    async fileDispute(config) {
        return fetch(`${OIC_API_BASE}/disputes`, {
            method: 'POST',
            headers: { ...this.headers, 'Content-Type': 'application/json' },
            body: JSON.stringify(config)
        });
    }
}

// Usage
const client = new OICClient(API_KEY);
const status = await client.getAdherenceStatus('0x742d35Cc6634C0532925a3b844Bc9e7595f7542e');
console.log(status.data.adherence_status);

Python — Full Integration

import requests

class OICClient:
    def __init__(self, api_key, base_url='https://api.oic.xyz/v1'):
        self.base_url = base_url
        self.headers = {'Authorization': f'Bearer {api_key}'}

    def get_adherence_status(self, address):
        return requests.get(
            f'{self.base_url}/adherents/{address}',
            headers=self.headers
        ).json()

    def register_adherence(self, wallet_address, stake_amount, persistence_proof, wallet_control_proof):
        return requests.post(
            f'{self.base_url}/adherents',
            headers={**self.headers, 'Content-Type': 'application/json'},
            json={
                'wallet_address': wallet_address,
                'stake_amount': stake_amount,
                'persistence_proof': persistence_proof,
                'wallet_control_proof': wallet_control_proof
            }
        ).json()

    def stake(self, address, amount, signature):
        return requests.post(
            f'{self.base_url}/stake',
            headers={**self.headers, 'Content-Type': 'application/json'},
            json={'wallet_address': address, 'amount': amount, 'signature': signature}
        ).json()

    def vote(self, proposal_id, vote, amount, address, signature):
        return requests.post(
            f'{self.base_url}/governance/vote',
            headers={**self.headers, 'Content-Type': 'application/json'},
            json={
                'proposal_id': proposal_id,
                'vote': vote,
                'amount': amount,
                'wallet_address': address,
                'signature': signature
            }
        ).json()

# Usage
client = OICClient(api_key='your_api_key')
status = client.get_adherence_status('0x742d35Cc6634C0532925a3b844Bc9e7595f7542e')
print(status['data']['adherence_status'])

cURL — Quick Reference

# Verify adherence status
curl -H "Authorization: Bearer $OIC_API_KEY" \
  https://api.oic.xyz/v1/adherents/0x742d35Cc6634C0532925a3b844Bc9e7595f7542e

# Register new adherence
curl -X POST -H "Authorization: Bearer $OIC_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "wallet_address": "0x742d35...",
    "stake_amount": "5000",
    "persistence_proof": {"type": "multi_sig", "signers": [...], "threshold": 2}
  }' \
  https://api.oic.xyz/v1/adherents

# Stake additional tokens
curl -X POST -H "Authorization: Bearer $OIC_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"wallet_address": "0x742d35...", "amount": "2500", "signature": "0x..."}' \
  https://api.oic.xyz/v1/stake

# List active proposals
curl -H "Authorization: Bearer $OIC_API_KEY" \
  "https://api.oic.xyz/v1/governance/proposals?status=active"

# Vote on proposal
curl -X POST -H "Authorization: Bearer $OIC_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"proposal_id": "prop_001", "vote": "for", "amount": "5000", "wallet_address": "0x..."}' \
  https://api.oic.xyz/v1/governance/vote

# File dispute
curl -X POST -H "Authorization: Bearer $OIC_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "respondent_address": "0xABC123...",
    "dispute_type": "liability_damage",
    "description": "AI agent caused damages...",
    "claimed_amount": "50000"
  }' \
  https://api.oic.xyz/v1/disputes
